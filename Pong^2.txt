int collide(balltype a, playertype b) {
	
	int aa, bb, c, d, e, f, g, h;


	aa = a.x - a.w;
	bb = a.x;
	c = a.y - a.h;
	d = a.y;

	e = b.x - b.w;
	f = b.x;
	g = b.y - b.h;
	h = b.y;
	

	if (aa > bb) {
		return 0;
	}

	if (c < d) {
		return 0;
	}

	if (e > f) {
		return 0;
	}

	if (g < h) {
		return 0;
	}

	return 1;
}

int collide2(balltype a, playertype b) {

	int aa, bb, c, d, e, f, g, h;


	aa = a.x - a.w;
	bb = a.x;
	c = a.y - a.h;
	d = a.y;

	e = b.x - b.w;
	f = b.x;
	g = b.y - b.h;
	h = b.y;
	

	if (aa > bb) {
		return 0;
	}

	if (c < d) {
		return 0;
	}

	if (e > f) {
		return 0;
	}

	if (g < h) {
		return 0;
	}

	return 1;
}

static void move_ball() {
	
	int c,d;
	for(int i =  0; i < 2; i++){
	ball[i].x += ball[i].speedx;
	ball[i].y += ball[i].speedy;
	
	if (ball[i].x < 0 ) {
		scores[0]++;
		reset = 0;
		if(scores[0] == 5)
			gameover = 0;
	}

	if (ball[i].x > (128-ball[i].w)) { 
		scores[1]++;
		reset = 0;
		if(scores[1] == 5)
			gameover = 0;
	}

	if (ball[i].y < ball[i].h || ball[i].y > (160-ball[i].h)) {
		
		ball[i].speedy= -ball[i].speedy;
	}

		c = collide(ball[i], player[1]); 
	  d = collide2(ball[i], player[0]);

		if (c == 1) 
			
			ball[i].speedx = -ball[i].speedx;
			
			if(d == 1)
			ball[i].speedx = -ball[i].speedx;
	}
}


void SysTick_Init(void){
	systickCount = 0;
	semaphore = 0;
  NVIC_ST_CTRL_R = 0;         // disable SysTick during setup
  NVIC_ST_RELOAD_R = 1000000;// reload value
  NVIC_ST_CURRENT_R = 0;      // any write to current clears it
  NVIC_SYS_PRI3_R = (NVIC_SYS_PRI3_R&0x00FFFFFF)|0x20000000; // priority 1
  NVIC_ST_CTRL_R = 0x0007; // enable SysTick with core clock and interrupts
}


void SysTick_Handler(void){
	semaphore=1; //perform an action
	systickCount++;
	if((GPIO_PORTE_DATA_R & 0x08) == 8)
				move_player(0);
	else if((GPIO_PORTE_DATA_R & 0x04) == 4)
				move_player(1);
	data =  ADC_In();
	if((temp-5)> data)
		move_player2(1);
	else if ((temp+5) < data)
		move_player2(0);
	temp = data;
}

}
